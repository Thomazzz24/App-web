mirar los clientes que hay activos 

SELECT 
    nombre_1,
    apellido_1,
    estado_cliente
FROM cliente c
INNER JOIN estado_cliente ec
    ON c.id_estado_cliente = ec.id_estado_cliente
WHERE c.id_estado_cliente = 1;
